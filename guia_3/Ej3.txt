/*
Dado un valor decimal ingresado por teclado, imprimir su valor binario equivalente como una
secuencia de ceros y unos.

*/
	MOV [AC],0
	MOV EDX,DS
	MOV EAX,0x01
	LDL ECX,1
	LDH ECX,4
	SYS 1
	
	MOV [4],[EDX]
sigue:	CMP [EDX],0
	JZ  fin
	SHR [EDX],1
	ADD [AC],1
	JMP sigue

	
otro:	MOV [EDX],[4]
	CMP [AC],0
	JZ stop
	MOV [8],[AC]
	SUB [8],1
	
	SHR [EDX],[8]
	SUB [AC],1
	
	AND [EDX],0x1
	MOV EAX,0x02
	LDL ECX,1
	LDH ECX,4
	SYS 2
	JMP otro
stop:
	stop
