otro:
	CMP [EDX],EBX

primo:
	MOV EDX,DS
	MOV [EDX],1
	MOV EAX,0x01
	LDL ECX,1
	LDH ECX,4
	SYS 2
	JMP stop
noprimo:
	MOV EDX,DS
	MOV [EDX],0
	MOV EAX,0x01
	LDL ECX,1
	LDH ECX,4
	SYS 2

stop:
